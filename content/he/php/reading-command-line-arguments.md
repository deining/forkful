## מדוע

קריאת ארגומנטים מהשורת פקודה היא כלי חשוב לפיתוח ב PHP ומאפשרת למפתחים לתכנן את קודםם בצורה יעילה יותר. היא מאפשרת לנו לקבל מידע מגוון על השטח הסביבתי שבו הקוד יפעל וכך לתכנן פעולות בקוד שמתאימות למידע המתקבל. 

## איך להשתמש

הקוד הבא מדגים כיצד לקרוא את הארגומנטים שמועברים משורת הפקודה ב-PHP ולהציג אותם בפורמט נוח לקריאה. 

```PHP
// המתודה הבאה קוראת את כל הארגומנטים המועברים בשורת הפקודה ומציגה אותם בפורמט של מערך 
function printCommandLineArguments(){
    $arguments = $_SERVER['argv']; // אוסף של כל הארגומנטים המועברים משורת הפקודה
    foreach($arguments as $argument){
        echo $argument . "\n"; // הדפסת כל ארגומנט בשורה נפרדת
    }
}

// קריאה למתודה המייצגת את הארגומנטים
printCommandLineArguments();

// פלט יתאים לקוד הבא: 
// php script.php argument1 argument2 argument3

// פלט הקוד 
// argument1
// argument2
// argument3
```
