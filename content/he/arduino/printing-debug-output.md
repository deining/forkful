## מדוע: 

למה ליצור מאמר על שפת תכנות ארדוינו סליחה? מספר משפטים אחד שניים שמסבירים למה משתמשים בהדפסת פלט לתקלות בקוד.

ההדפסה של פלט לתקלות היא כלי חשוב ושימושי בתהליך הפיתוח של קוד ארדוינו. כאשר אנחנו מדפיסים מידע לתקלות, אנחנו מקבלים מידע חיוני על פעולת הקוד ואיפה יש לנו בעיות. הנה כמה דוגמאות של תקלות שאולי יכולות להיכנס למאמר זה ולהיראות:

```Arduino
void setup() {
  // כאן אנחנו מדפיסים את הפלט הראשוני שלנו להתחלה
  Serial.begin(9600);
  Serial.println("פלט ראשוני - אנחנו מתחילים");
}
```

```Arduino
void loop() {
  // כאן אנחנו מדפיסים נתונים חשובים כמו זמן וערך מתאים של חיישן
  Serial.print("התקבל נתונים מהחיישן בזמן: ");
  Serial.print(millis());
  Serial.print(" ערך הנתונים הוא: ");
  Serial.println(sensorValue);
  // כאן אנחנו יכולים לראות את הנתונים שמתקבלים בזמן אמת ולבדוק את הערכים שלנו
}
```
