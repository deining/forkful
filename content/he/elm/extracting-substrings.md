# למה

ניתן לכתוב טקסט רב עוצמה תחת כותרת מסוימת, אך נדיר למצוא את הטקסט הנדרש מראש עד סופו. ניתן להשתמש במתודות שונות כמו החיתוך (substring) ליצירת תמצית טקסט רק על סמך נקודות רצויות, כדי להגיע למסקנות בו. את התוצאות האלה ניתן להשתמש בהם לסיכום, להדגמת היכולות או למתוח את הטקסט הקיים.



הוא רק מכיל את שורות הטקסט הקשורות לחיתוך שנראה בתוך התוויות "``` Elm ...```: 

## איך לעשות זאת

הנה מספר דוגמאות מפורטות בהן קוד ה-Elm נמצא בשתי לשוניות: חשיפה (exposure) ותוצאות. דוגמא ראשונה בה המשתמש מסמן את הטקסט הנדרש דרך תו מסוים:

``` Elm 
-- חיתוך טקסט באמצעות (substring)
input = "אני רוצה לכתוב ב-Elm"

אם נרצה לכתוב טקסט את מילה "רוצה" בלבד, נستخدם פונקציית החיתוך ונפריד את הטקסט לסעיפיו.

exposure = "רוצה"

-- תוצאות
wanted = String.slice 2 5 input
```

דוגמא אחרת בה אווירה את הפונקציות השלישיות, בהן מתאימים את הגבולות של הטקסט שתורכלו להדמיין:

``` Elm
-- חיתוך טקסט באמצעות (substring)
text = "אני מתאהב ב-Elm"

אם נרצה לכתוב את המילה הראשונה בכותרת "מתאהב" בלבד, ניתן לכתוב את זה גם כן כפי שנראה הליך הכתיבה הבא:

subtitle = "מתאהב ב"
-- פונקציית חיתוך היא line, ניתן לקבל את תוצאותיו בשר הכריכה שלו.
-- יש לנו גבולות של הטקסט
translation = String.slice 6
