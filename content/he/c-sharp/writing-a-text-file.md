# למה

כתיבת קובץ טקסט היא דבר הדורש המון מאמץ ועבודה, אבל התשואות שמתקבלות מכך הן מתוקות מאוד. בכתיבת קובץ טקסט בשפת C# יש את היתרונות של יכולות התוכנה במלואן וזהו כדי שחלופות מאוחרות בהכתבה תהיה אפשרית לנו כבר מתוך ההתחלה.

# איך לעשות

הנה דוגמא פשוטה לכתיבת קובץ טקסט בשפת C#:

```C#
//קוד כדוגמא לכתיבת קובץ טקסט
using System;
using System.IO;

class Program
{
    static void Main()
    {
        //מציאת או יצירת קובץ טקסט חדש
        string filePath = "C:\\myTextFile.txt";
        File.Create(filePath).Dispose();

        //כתיבת תוכן לקובץ טקסט
        string[] lines = { "שלום", "זהו קובץ טקסט מוצלח!", "מחשבנו משתמש כאן לטובת חישוב מספרים.", "תודה על הקריאה!" };
        File.WriteAllLines(filePath, lines);

        //שליחת התוכן למחשב על מנת לבדוק אם הכתיבה בוצעה בהצלחה
        Console.WriteLine($"קובץ הטקסט כתוב בדרך כלל כדור שמחובר למחשב: {filePath}");
        Console.WriteLine("פעמים רבות לא ייהיה תאימות עם סוג המסך המצוין שבבית שלנו");
        // אם סוג המסך לא מתאים, ישים לב שכתיבת הקובץ תצטרך התאמה מלאה
        Console.ReadLine();
    }
}
```

וכך נראה הפלט של התכנית:

```
קובץ הטקסט כתוב בדרך כלל כדור שמחובר למחשב: C:\\myTextFile.txt
פעמים רבות לא ייהיה תאימות עם סוג המסך המצוין שבבית שלנו
```
