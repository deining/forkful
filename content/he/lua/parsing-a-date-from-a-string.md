## מה זה ולמה?
כשאתה מכניס תאריך למחשב, הוא יכול לברר מאיזה תאריך אתה מדבר, אם אתה מחפש לבדוק את הנתונים, או אם אתה רוצה לייצר תאריך חדש באופן אוטומטי. פילוח תאריך ממחרוזת הוא פעולה שתעזור לך לקבל את התאריך המדויק שאתה מחפש, בלי להיכנס לפרטים של צבעים או שעות. תכניתנים משתמשים גם בפילוח תאריך ממחרוזת כדי להפוך את המידע הזה לנתוני תאריך נורמליים שניתן לעבוד איתם בקלות.

## איך לעשות את זה:
נותנים לך דוגמאות בקוד בלועא, כדי שתוכל לראות מה התוכנית תציג כשאתה מזין את המחרוזת הרלוונטית.

```lua
-- דוגמה ראשונה, מחרוזת של תאריך קונספטואלי
local date = "2019-12-25"

-- מפרקים את התאריך לפי המחרוזת
year, month, day = date:match("(%d%d%d%d)-(%d%d)-(%d%d)")

-- מדפיסים את התאריך החדש בפורמט נורמלי
print(day .. "/" .. month .. "/" .. year)
```

תוכל לראות שהתחום "year, month, day" נמוך במחרוזת למעלה. אנחנו משתמשים בפונקציית פרק לקבלת התאריך המדויק שאנחנו רוצים. זה יוצר תאריכים בפורמט דייטווז, בסדר ספציפי שתוכל לעבוד בקלות עם המידע הזה.

```lua
-- דוגמה שניה, מחרוזת של תאריך אנגלי
local date = "December 25, 2019"

-- מפרקים את התאריך לפי המחרוזת
month, day, year = date:match("(%a+) (%d+), (%d+)")

-- מדפיסים את התאריך החדש בפורמט נורמלי
print(day .. "/" .. month .. "/" .. year)
```

בדוגמה הזאת, אנחנו מסתמכים על הפונקציות "אחרי, חמישי, שנה" כי הם מתאימים לפורמט המקובל לתאריך אנגלי, וכמובן, אתה יכול לשנות את סדר האלמנצים למה שרצית. עסקנו גם בתווי אחוז שנתיים, שהופכים לא חוקיים כשעוברים מן המחרוזת.
