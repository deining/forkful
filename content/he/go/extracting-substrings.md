## למה

למה להתעסק עם חילוץ תת-מחרוזת? התשובה פשוטה - כי זה כלי חיוני בתכנות בשפת גו! חילוץ תת-מחרוזת מאפשר לנו להבין ולעבוד עם מחרוזות בצורה מדויקת יותר, ומאפשר לנו לתכנת בצורה יעילה יותר כאשר אנו מתבטאים בעזרת תנאים ולולאות.

## כיצד לעשות זאת

להלן דוגמאות של כיצד לחלץ תת-מחרוזת בעזרת קוד גו:

```Go
// תהליך חליצת תת-מחרוזת מחלקת המחרוזת לתאים ומחזירה את התא המתאים
func extractSubstring(str string, start int, length int) string {
    // חילץ את התא הרצוי בעזרת הפונקציה "substring" של שפת גו
    return str[start:start+length]
}

// הדפסת תת-מחרוזת מחלקת המחרוזת לתאים ומדפיסה את התא המתאים
func printSubstring(str string, start int, length int) {
    // הדפס את התא הרצוי בעזרת הפונקציה "substring" של שפת גו
    fmt.Printf("תת-מחרוזת המתחילה בתא %v ומכילה %v תווים היא: %v\n", start, length, str[start:start+length])
}

func main() {
    // חילוץ תת-מחרוזת באינדקס 2 ואורך 5 תווים ממחרוזת "זאת המחרוזת"
    sub := extractSubstring("זאת המחרוזת", 2, 5)
    fmt.Println(sub) // תמחר

    // הדפסת תת-מחרוזת באינדקס 1 ואורך 4 תווים ממחרוזת "זאת המחרוזת"
    printSubstring("זאת המחרוזת", 1, 4) // המחר

}
```

כפי שאפשר לראות, חילוץ תת-מחרוזת בעזרת קוד גו פשוט ויעיל. באמצעות קוד זה, אנו מגלים תת-מחרוזות מתוך מחרוזות גדולות יותר, נקודת התחלה משמשת לקביעת איזה תת-מחרוזת אנו רוצים לחלץ ואורך התת-מחרוזת קובע את כמות
