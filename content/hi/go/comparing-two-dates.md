"Why"
क्यों: दो तारीखों को तुलना करने में किसी को लिये क्यों शामिल होना चाहिए।

हर किसी के जीवन में तारीखों का महत्व होता है। यह अपने जन्मदिन, शादी की तारीख, या किसी अन्य खुशी के मौके की तारीख हो सकती है। अगर हम दो तारीखों को तुलना करने में सक्षम होते हैं, तो हम अपने जीवन के महत्वपूर्ण घटनाओं की शुभकामनाएं दे सकते हैं या समय के साथ आने वाली घटनाओं को पता लगा सकते हैं।

"How To"
कैसे करें: कोडिंग उदाहरण और "```Go ... ```" कोड ब्लॉक के भीतर नमूना उत्पादन।

गो प्रोग्रामिंग भाषा में दो तारीखों को तुलना करने के लिए आपको पहले से तैयार महत्वपूर्ण फ़ंक्शन्स हैं। इन फ़ंक्शन्स का उपयोग करके, आप दो तारीखों को तुलना और तारीखों के बीच अंतर को पास्त कर सकते हैं। नीचे दिए गए कोड सामग्री उदाहरण में आपको इसका उपयोग कैसे करना है वह दिखाया गया है।

```Go
// दो तारीखों को तुलना करने के लिए फ़ंक्शन
func CompareDates(date1, date2 time.Time) int {
    // तारीखों के बीच अंतर को पास करें
    return date1.Sub(date2).Hours()
}

func main() {
    // दो तारीखों बनाएं
    date1 := time.Date(2021, 1, 1, 0, 0, 0, 0, time.UTC)
    date2 := time.Date(2021, 1, 10, 0, 0, 0, 0, time.UTC)

    // फ़ंक्शन को कॉल करें और परिणाम प्रिंट करें
    result := CompareDates(date1, date2)
    fmt.Println(result) // output: -216

    // अंतर के साथ इस तारीख को भी पास कर सकते हैं
