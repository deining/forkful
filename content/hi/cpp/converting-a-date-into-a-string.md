## क्यों: 
क्या आपने कभी सोचा है कि अपने प्रोग्राम में तिथि को स्ट्रिंग में कन्वर्ट करने का फायदा क्यों हो सकता है? यह एक बहुत ही उपयोगी टेक्निक है जो आपको अपने कोड को लंबे और जटिल कोड से बचाकर आसानी से संभालने में मदद कर सकता है।

## कैसे करें: 
यदि आप C++ प्रोग्रामिंग में तारीख को स्ट्रिंग में कन्वर्ट करना चाहते हैं, तो आप निम्नलिखित कोड का उपयोग कर सकते हैं।

```C++
// कक्षा को निर्धारित करें
#include <iostream>
#include <sstream>
#include <string>
#include <ctime>

using namespace std;

int main()
{
  // प्रारंभ तिथि को सेट करें
  int दिन = 31;
  int महीने = 12;
  int साल = 2021;

  // समय को सेट करें
  struct tm t = {0};
  t.tm_mday = दिन;
  t.tm_mon = महीने - 1; // 0 जनवरी, 1 दिसंबर से शुरू होता है
  t.tm_year = साल - 1900; // 1900 से पहले के साल गणितीय वर्ष होते हैं

  // समय को स्ट्रिंग में कन्वर्ट करें
  stringstream ss;
  ss << put_time(&t, "%d/%m/%Y");
  string तिथि = ss.str();

  // स्ट्रिंग तिथि को प्रिंट करें
  cout << "आपकी तिथि: " << तिथि << endl;

  // सत्यापन से पूर्व तिथि को प्रिंट करें
  string सत्यापन = (तिथि == "31/12/2021") ? "सफलतापूर्वक" : "असफल";
  cout << "तिथि कन्वर्टिंग " << सत्यापन << " हुई।" << endl;
  return 0;
}
```
आउटपुट:
```
आपकी तिथि: 31/12/2021
तिथि कन्वर्टिंग सफलतापूर्वक हुई।
```
