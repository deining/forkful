## How to: (कैसे करें:)
```C++
#include <iostream>
#include <algorithm>
#include <string>

int main() {
    std::string data = "H3llo W@rld!";
    const std::string pattern = "1234567890@!";

    data.erase(std::remove_if(data.begin(), data.end(),
                              [&](char ch) { return pattern.find(ch) != std::string::npos; }),
               data.end());

    std::cout << data << std::endl; // Output: Hello Wrld
    return 0;
}
```
