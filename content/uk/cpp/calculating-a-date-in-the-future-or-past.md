## Чому

По можливості на веселощі, щоб забавитися або практикуватися, для вивчення нового алгоритму чи просто для цікавості - є багато причин, чому люди можуть використовувати програмування для обчислення дат у майбутньому або минулому.

## Як виконати

```C++
#include <iostream>
using namespace std;

// Функція, яка обчислює дату у майбутньому
void дата_майбутнього(const int дні, const int місяці, int роки)
{
    // Виведення поточної дати
    cout << "Поточна дата: " << дні << "/" << місяці << "/" << роки << endl;

    // Обчислення дати у майбутньому, додаючи до поточної дати задану кількість днів
    int нова_дата = дні + 10;
    int новий_місяць = місяці;
    int новий_рік = роки;

    // Якщо кількість днів перевищує 31, то додаємо новий місяць
    if (нова_дата > 31)
    {
        новий_місяць++;
        // Якщо новий місяць перевищує 12, то додаємо новий рік
        if (новий_місяць > 12)
        {
            новий_рік++;
            новий_місяць = 1;
        }
        // Віднімаємо 31 (максимальна кількість днів у місяці) від нової дати
        нова_дата = нова_дата - 31;
    }
    // Виведення нової дати у майбутньому
    cout << "Дата у майбутньому: " << нова_дата << "/" << новий_місяць << "/" << новий_рік << endl;
}
