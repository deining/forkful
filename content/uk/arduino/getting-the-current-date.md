## Як це зробити

Щоб отримати поточну дату, необхідно підключити реальний часовий модуль RTC (Real Time Clock) до плати Arduino і скористатися вбудованими функціями бібліотеки RTClib. Давайте познайомимося із використанням цієї бібліотеки за допомогою наступного коду:

```Arduino
#include <RTClib.h> // підключення бібліотеки роботи з реальним часом
 
RTC_DS1307 rtc; // створення об'єкту для роботи з RTC
 
void setup(){ // початок програми
	Serial.begin(9600); // встановлення швидкості передачі даних за сучасним протоколом
	while (!Serial) { } // очікування на доступність порту
	if (! rtc.begin()) { // спроба підключення до модуля
		Serial.println(F("На модулі пошкоджений годинник COD DS1307
Підключіть модуль до плати Arduino!"));
	}		
}
 
void loop(){ // головний цикл програми
	DateTime now = rtc.now(); // отримання поточної дати із модуля
	Serial.print(now.year(), DEC); // виведення на друк цілого числа
	Serial.print('/'); // виведення символу
	Serial.print(now.month(), DEC); // виведення на друк цілого числа
	Serial.print('/'); // виведення символу
	Serial.print(now.day(), DEC); // виведення на друк цілого числа
	Serial.print(' '); // виведення символу
	Serial.print(now.hour(), DEC); // виведення на друк цілого числа
	Serial.print(':'); // виведення символу
	Serial.print(now.minute(), DEC); // виведення на друк цілого числа
	Serial.print(':'); // виведення символу
	Serial.print(now.second(), DEC); // виведення на друк цілого числа
	Serial.println(); // виведення рядка
	delay(1000); // затримка мілісекунди
}
```

Після завантаження програми до плати, ви маєте побачити на консолі поточну дату та час, що виводиться в такому форматі: "рік/місяць/день година:хвилина:секунда".
